```c++
#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int c=0,a[50005]={0},sum=0;

int maxx(int l,int r)
{
    c++;
    if(l==r)
    {
        if(a[l]<0)return 0;
        else return a[l];
    }
    else
    {
        int mid=(l+r)/2;
        int lmax=maxx(l,mid);
        int rmax=maxx(mid+1,r);
        int s=0;
        for(int i=l;i<=r;i++)
        {
            if(s+a[i]>0)
            {
                s=s+a[i];
                sum=max(sum,s);
            }
            else s=0;
        }
        sum=max(lmax,sum);
        sum=max(rmax,sum);
    }
    return sum;
}

int main()
{
    ios::sync_with_stdio(false);
    int n,i;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    cout<<maxx(0,n-1)<<" "<<c<<endl;
    return 0;
}
```
